MAIN CLASS-----------------
import java.util.ArrayList;
import java.util.Scanner;

/**
 * Created by kverma on 5/18/2017.
 */
public class PersonalDetailsMain
{
	static void printData(ArrayList<PersonalDetailsBean> dataList)
	{// Print the data
		for (PersonalDetailsBean datalst : dataList)
		{
			System.out.print("Id :" + datalst.getId() + "\n");

			System.out.print("Name :" + datalst.getName() + "\n");

			System.out.print("Address :" + datalst.getAddress() + "\n");

			System.out.print("Phone Number :" + datalst.getPhone() + "\n");

		}

	}

	public static void main(String args[])
	{
		Scanner scanner = new Scanner(System.in);
		System.out.println("Total No of People : ");
		int totalCount = Integer.parseInt(scanner.nextLine());

		ArrayList<PersonalDetailsBean> dataList = new ArrayList<PersonalDetailsBean>();
		// Read all the data from User
		for (int i = 0; i < totalCount; i++)
		{
			PersonalDetailsBean bean = new PersonalDetailsBean();
			// id can also be autogenerated based on value of i
			System.out.print("Id :");
			bean.setId(scanner.nextInt());

			// Get the user's Name.
			System.out.print("Name :");
			bean.setName(scanner.next());

			// Get the user's address.
			System.out.print("Address :");
			bean.setAddress(scanner.next());

			// Get the user's phone number.
			System.out.print("Phone Number :");
			bean.setPhone(scanner.next());
			dataList.add(bean);
		}
		printData(dataList);

	}
}

BEAN CLASS--------------------------------------------
/**
 * Created by kverma on 5/18/2017.
 */
public class PersonalDetailsBean
{

	private int id;
	private String name;
	private String phone;
	private String address;

	public int getId()
	{
		return id;
	}

	public void setId(int id)
	{
		this.id = id;
	}

	public String getName()
	{
		return name;
	}

	public void setName(String name)
	{
		this.name = name;
	}

	public String getPhone()
	{
		return phone;
	}

	public void setPhone(String phone)
	{
		this.phone = phone;
	}

	public String getAddress()
	{
		return address;
	}

	public void setAddress(String address)
	{
		this.address = address;
	}
}
